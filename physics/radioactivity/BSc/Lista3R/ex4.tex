\begin{exercicio}[blue]
	Discuta as diferenças e semelhanças entre as interações de partículas carregadas e as interações
	de fótons gama e X com a matéria.
\end{exercicio}

\begin{minipage}[t]{0.46\textwidth}
	\centering
	\center{\sc PARTÍCULAS CARREGADAS}
	\begin{itemize}
	\item	{\bf sempre} interagem com o alvo.
	\item A maioria dos efeitos ocorre por interações Coulombianas com transferência parcial
				(predominante) ou total da energia.
	\item	Pode causar ionização e excitação no meio.
	\item	A trajetória do feixe emergente em geral difere bastante da do feixe incidente para 
			  partículas leves (elétrons ou pósitrons); e pouco para partículas pesadas (prótons,
			  núcleos e átomos).
	\item	Raidações bariônicas (prótons, núcleons) e leptônicas (elétrons e pósitrons) geram radiação 					fotônica.
	\item Conserva energia mas pode não conservar momento.
	\end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.46\textwidth}
	\centering
	\center{\sc FÓTONS}
	\begin{itemize}
	\item {\bf podem} interagir ou não com o alvo.
	\item A maioria dos efeitos ocorre por choque, com absorção parcial ou total da energia dos fótons
				por parte dos elementos do alvo.
	\item Pode causar ionização e excitação no meio.
	\item	A trajetória do feixe emergente em geral difere bastante da do feixe incidente.
	\item Radiação fotônica gera radiações bariônicas e leptônicas.
	\item	Conserva momento e energia.
	\end{itemize}
\end{minipage}
