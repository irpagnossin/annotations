\begin{quotation}
{\bf 12.}
{\it	Calcule o comprimento de onda e a energia dos fótons para as três primeiras transições das séries
	de Lyman, Balmer e Paschen do átomo de Hidrogênio.}
\end{quotation}

	A expressão que relaciona o comprimento de onda irradiado (ou absorvido) numa transição 
eletrônica do átomo de Hidrognio é
	$$
	{1\over\lambda}=R({1\over n_f^2}-{1\over n_i^2})
	$$
	
	onde $R\cong10968100\,\,\mathrm{m^{-1}}$ é a constante de Rydberg, $n_i$ e $n_f$ são
os números quânticos principais inicial e final, respectivamente. 

	Para a série de Lyman sabemos que $n_f=1$. Logo,
	\begin{eqnarray*} 
	E(1,2) &\rightarrow& {1\over\lambda_{12}}=R(1-{1\over4}) \Rightarrow \lambda_{12}=
							1215\,\,\mbox{\AA}\,\,(8,6\,\,\mathrm{eV}) \\
	E(1,3) &\rightarrow& {1\over\lambda_{13}}=R(1-{1\over9}) \Rightarrow \lambda_{13}=
							1026\,\,\mbox{\AA}\,\,(10,2\,\,\mathrm{eV}) \\
	E(1,4) &\rightarrow& {1\over\lambda_{14}}=R(1-{1\over16}) \Rightarrow \lambda_{14}=
							972\,\,\mbox{\AA}\,\,(10,7\,\,\mathrm{eV}) 
	\end{eqnarray*}

	Para a série de Balmer $n_f=2$ e, portanto,
	\begin{eqnarray*}
	E(2,3) &\rightarrow& {1\over\lambda_{23}}=R({1\over4}-{1\over9}) \Rightarrow \lambda_{23}=
							6564\,\,\mbox{\AA}\,\,(1,6\,\,\mathrm{eV}) \\
	E(2,4) &\rightarrow& {1\over\lambda_{24}}=R({1\over4}-{1\over16}) \Rightarrow \lambda_{24}=
							4862\,\,\mbox{\AA}\,\,(2,1\,\,\mathrm{eV}) \\
	E(2,5) &\rightarrow& {1\over\lambda_{25}}=R({1\over4}-{1\over25}) \Rightarrow \lambda_{25}=
							4341\,\,\mbox{\AA}\,\,(2,4\,\,\mathrm{eV}) 
	\end{eqnarray*} 

	Para a série de Paschen $n_f=3$,
	\begin{eqnarray*}
	E(3,4) &\rightarrow& {1\over\lambda_{34}}=R({1\over9}-{1\over16}) \Rightarrow \lambda_{34}=
							18755\,\,\mbox{\AA}\,\,(0,5\,\,\mathrm{eV}) \\
	E(3,5) &\rightarrow& {1\over\lambda_{35}}=R({1\over9}-{1\over25}) \Rightarrow \lambda_{35}=
							12821\,\,\mbox{\AA}\,\,(0,8\,\,\mathrm{eV}) \\
	E(3,6) &\rightarrow& {1\over\lambda_{36}}=R({1\over9}-{1\over36}) \Rightarrow \lambda_{36}=
							10941\,\,\mbox{\AA}\,\,(0,9\,\,\mathrm{eV}) 
	\end{eqnarray*}
