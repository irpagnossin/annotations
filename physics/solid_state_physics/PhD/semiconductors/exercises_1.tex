	\textbf{Exercício 1:} Considere o operador de rotação $C_\theta$ numa família de planos $(m_1m_2m_3)$ --- onde $m_i$ são índices de Miller --- dado por
	$$
	C_n = \left(
	\begin{matrix}
		\cos\theta	&	-\sin\theta	\\
		\sin\theta	&	\cos\theta
	\end{matrix}\right),
	$$
	sendo $\theta$ o ângulo de rotação. Tome agora um vetor $\vetor R = (x,y,z)$ da rede direta que pertença ao plano $(m_1m_2m_3)$ e apliquemo-lhe $C_\theta$:
	$$
	C_\theta\,\vetor R = 
	\left(
	\begin{matrix}
		\cos\theta	&	-\sin\theta	\\
		\sin\theta	&	\cos\theta
	\end{matrix}\right) \cdot
	\left(\begin{matrix}
	x \\
	y \\
	z
	\end{matrix}\right) = 
	\left(\begin{matrix}
		x\,\cos\theta-y\,\sin\theta	\\
		x\,\sin\theta+y\,\sin\theta
	\end{matrix}\right).
	$$
	
	Por outro lado,
	$$
	C_{-\theta}\,\vetor R = 	
	\left(\begin{matrix}
		x\,\cos\theta+y\,\sin\theta	\\
		-x\,\sin\theta+y\,\sin\theta
	\end{matrix}\right),
	$$
	de modo que $(C_\theta + C_{-\theta})\,\vetor R = 2\,\cos\theta\,\vetor R$. Assim, $C_\theta + C_{-\theta} = \lambda \in Z$ e podemos escrever $\lambda = 2\,\cos\theta$. Como $\theta\in[0,2\,\pi]$, o valor máximo (mínimo) de $\lambda$ é $2$ ($-2$). Entrementes, $\lambda\in Z$ e, por conseguinte, é igual a $-2$, $-1$, $0$, $1$ ou $2$. Finalmente, se escrevermos $\theta$ como uma fração de $2\,\pi$, ou seja, como $2\,\pi/n$ ($n\in Z$), concluimos que
	$$
	n = \frac{2\,\pi}{\arccos\left(\lambda/2\right)}
	$$
	e decorre dos valores possíveis de $\lambda$ que $n=1,2,\ldots,6$.

	\textbf{Exercício 6:} O fator de estrutura procurado é 
	$$
	S_{\vetor k} = \sum_j f_j\,e^{i\vetor k\cdot \vetor R_j} = f_0 \sum_j e^{i\vetor k\cdot \vetor R_j},
	$$
	onde $f_j=f_0$ é o fator de estrutura atômico, igual para todos os átomos de carbono da célula convencional. Os vetores $\vetor R_j$ localizam os átomos da base utilizada e, no nosso caso, são oito (em termos dos versores ortonormais canônicos):

	\begin{align*}
	\vetor R_1 &= (0,0,0)							&\qquad	\vetor R_5 &= \frac{a}{4}(1,1,1)	\\
	\vetor R_2 &= \frac{a}{2}(0,1,1)	&\qquad	\vetor R_6 &= \frac{a}{4}(3,1,3) \\
	\vetor R_3 &= \frac{a}{2}(1,0,1)	&\qquad	\vetor R_7 &= \frac{a}{4}(1,3,3) \\
	\vetor R_5 &= \frac{a}{2}(1,1,0)	&\qquad	\vetor R_8 &= \frac{a}{4}(3,3,1)
	\end{align*}

	
	Fazendo $\vetor k = (h,k,l)$, onde $h$, $k$ e $l$ são índices de Miller, ficamos com
	\begin{multline*}
	\frac{S_{\vetor k}}{f_0} = 1 + e^{i\pi(h+k)} + e^{i\pi(k+l)} + e^{i\pi(l+h)} + e^{i\pi(h+k+l)/2} + \\ e^{i\pi(3h+3k+l)/2} + 
				e^{i\pi(3h+k+3l)/2} + e^{i\pi(h+3k+3l)/2},
	\end{multline*}
	que pode ser reorganizado da seguinte forma:
	\begin{align*}
	\frac{S_{\vetor k}}{f_0} &= \left[ 1 + e^{i\pi(h+k)} + e^{i\pi(k+l)} + e^{i\pi(l+h)} \right]\,\left[ 1 + e^{i\pi(h+k+l)/2} \right] \\
			& = \left[ 1 + (-1)^{h+k} + (-1)^{k+l} + (-1)^{l+h} \right]\left[ 1 + i^{h+k+l} \right]
	\end{align*}
	
	Os zeros do fator de estrutura ocorrem quando (I) $\exp\left[i\pi(h+k+l)/2\right]=-1$ \textbf{ou} (II) quando apenas um dos parênteses no colchete mais à esquerda é $+1$ e os outros, $-1$ (anulando todo o colchete). A primeira condição é verdadeira quando o argumento da exponencial é um múltiplo ímpar de $\pi$, ou seja, quando $h+k+l = 2(2n-1)$, $n=1,2,\ldots$. Por outro lado, a segunda condição exige que \textbf{apenas} uma das somas $h+k$, $k+l$ ou $l+h$ seja par (levando todo o colchete da esquerda a zero), enquanto as outras duas sejam ímpares, o que pode ser satisfeito quando temos uma mistura de números pares e ímpares (não todos pares nem todos ímpares). Assim, os zeros do fator de estrutura ocorrem quando
	\begin{equation}\label{eq:minimos}
	h+k+l = 2(2n-1), \quad n = 1,2,\ldots
	\end{equation}
	e quando $h$, $k$ e $l$ são uma mistura de pares e ímpares.
	
	A intensidade das reflexões de fótons incididos sobre a estrutura é aproximadamente proporcional a $|S_{\vetor k}|^2$:
	\begin{equation}\label{eq:reflexoes}
	|\frac{S_{\vetor k}}{f_0}|^2 = \left| 1 + (-1)^{h+k} + (-1)^{k+l} + (-1)^{l+h} \right|^2 \, \left| 1 + i^{h+k+l} \right|^2
	\end{equation}
	
	Assim, as reflexões possíveis ocorrem quando $|S_{\vetor k}|^2\ne 0$: Um máximo (absoluto ou relativo) de $|S_{\vetor k}|^2$ ocorre quando todos as parcelas no colchete da esquerda forem $+1$, ou seja, quando todos os expoentes forem pares. Isto ocorre sempre que os índices forem (iii) todos pares ou (iv) todos ímpares. Em (iii), podemos escrever $(h,k,l) = (2m_1,2m_2,2m_3)$, com $m_i=1,2,\ldots$. Neste caso,
	\begin{equation}\tag{iii}
	h + k + l = 2 (m_1 + m_2 + m_3) = 2m, \quad m = 1,2,\ldots
	\end{equation}
	
	Por outro lado, em (iv) podemos escrever $(h,k,l) = (2m_1-1,2m_2-1,2m_3-1)$e, por conseguinte,
	\begin{equation}\tag{iv}
	h + k + l = \underbrace{2 (m_1 + m_2 + m_3 + 1)}_{\text{par}} + 1 = 2m + 1, 	\quad m = 1,2,\ldots
	\end{equation}

	Agora precisamos ver o comportamento de $i^{h+k+l}$ na equação~\eqref{eq:reflexoes}: Quando valer~(iii) acima, teremos $i^{2m} = (-1)^m$. Portanto, para que o termo $|1+i^{h+k+l}|$ não se anule devemos impor que $m$ seja par,\footnote{De fato, comparando (iii) com \eqref{eq:minimos} veremos que se $m$ for ímpar, estaremos em um mínimo: $|S_{\vetor k}/f_0|=0$.} ou seja, que $m=2n$. E neste caso teremos 
	$$
	|\frac{S_{\vetor k}}{f_0}|^2 = |1+1+1+1|^2\,|1+1|^2 = 64.
	$$
	
	Por outro lado, quando valer~(iv), teremos $i^{h+k+l} = i^{2m+1} = i\,(-1)^m$ e o termo $|1+i^{h+k+l}|$ ficará igual a $|1\pm i|=\sqrt{2}$; e isto leva a
	$$
	|\frac{S_{\vetor k}}{f_0}|^2 = |1+1+1+1|^2\,(\sqrt{2})^2 = 32.
	$$
	
	Finalmente, concluimos as reflexões permitidas ocorrem quando $h+k+l=2(2n)$ (equação (iii) com $m=2n$) ou quando todos os índices forem ímpares (equação (iv), lembrando que uma mescla de pares com ímpares resulta em um mínimo).
	
	No caso da célula primitiva, onde a base é composta por apenas dois átomos em $\vetor R_1=(0,0,0)$ e $\vetor R_2=a(1,1,1)/4$,
	$$
	\frac{S_{\vetor k}}{f_0} = 1 + e^{i\pi(h+k+l)/2} = 1 + i^{h+k+l} .	
	$$
		
	Então, os zeros ocorrem quando $i^{h+k+l}=-1$, ou seja, quando $h+k+l=2(2n-1)$. As reflexões, por outro lado, ocorrem quando $|S_{\vetor k}|^2\ne0$, i.e., quando $i^{h+k+l}=+1$ ou $\pm i$. O primeiro caso ocorre quando $h+k+l = 2(2n)$; o segundo, quando $h+k+l = 2n+1$.
	
	Assim, embora a estrutura seja a mesma, os zeros e as condições de reflexão são diferentes em cada uma das descrições acima. Isto ocorre simplesmente por que a base utilizada é diferente: No primeiro caso, utilizamos oito átomos; no segundo, apenas dois. Contudo, é importante ressaltar que as descrições são equivalentes, pois referem-se a uma mesma estrutura.

	\textbf{Exercício 9:}	O método tight-bindig consiste em resolver o problema de autovalores $\mathcal{H}\ket{\psik}=E(\vetor k)\ket{\psik}$ com $\mathcal{H} = \mathcal{H}_\text{at} + \Delta U$, $\psik=\sum_{\vetor R} e^{i\vetor k\cdot \vetor R}\,\phi(\vetor r-\vetor R)$ e $\phi(\vetor r)=\sum_nb_n\,\psii{n}$, sendo $\psii{n}$ autofunções do problema atômico conhecido $\mathcal{H}_\text{at}\ket{\psi}=E(\vetor k)\ket{\psi}$. $\Delta U\ll \mathcal{H}_\text{at}$ é uma perturbação. Multiplicando o problema de autovalores por $\bra{\psii{n}}$, encontramos $\hope{\psii{n}}{\mathcal{H}}{\psik} = E(\vetor k)\iprod{\psii{m}}{\psik}$. Vamos resolver cada um dos membros desta equação:
	\begin{align*}
		\iprod{\psii{n}}{\psik} & = \sum_{n,\vetor R}b_n\,e^{i\vetor k\cdot\vetor R} \iprod{\psii{n}}{\psiiR{m}} \\
					& = \sum_n b_n\,\delta_{nm} + \sum_{n,\vetor R\ne\vetor 0}b_n\,e^{i\vetor k\cdot\vetor R} \iprod{\psii{n}}{\psiiR{m}} \\
					& = b_m  + \sum_{n,\vetor R\ne\vetor 0}b_n\,e^{i\vetor k\cdot\vetor R} \iprod{\psii{n}}{\psiiR{m}}.
	\end{align*}
	
	Além disso,
	\begin{align*}
		\hope{\psii{m}}{\mathcal{H}}{\psik}	&= \hope{\psii{m}}{\mathcal{H}_\text{at} + \Delta U}{\psik} \\
					& = \hope{\psii{m}}{\mathcal{H}_\text{at}}{\psik} + \hope{\psii{m}}{\Delta U}{\psik}\\
					& = \hope{\psik}{\mathcal{H}_\text{at}^\dagger}{\psii{m}}^* + \hope{\psii{m}}{\Delta U}{\psik}\\
					& = E_m \iprod{\psii{m}}{\psik} + \hope{\psii{m}}{\Delta U}{\psik},
	\end{align*}
	pois $E_m^*=E_m$ e $\mathcal{H}_\text{at}^\dagger = \mathcal{H}_\text{at}$.
	\begin{align*}
		\hope{\psii{m}}{\mathcal{H}}{\psik}	&= E_m \iprod{\psii{m}}{\psik} + \sum_{n,\vetor R} b_n\, e^{i\vetor k\cdot \vetor k} \, \hope{\psii{m}}{\Delta U}{\psiiR{n}} \\
			& = E_m \iprod{\psii{m}}{\psik} + \sum_n b_n\hope{\psii{m}}{\Delta U}{\psii{n}} +\\
				& \rule{2cm}{0pt} + \sum_{n,\vetor R\ne\vetor 0} b_n\, e^{i\vetor k\cdot \vetor k} \, \hope{\psii{m}}{\Delta U}{\psiiR{n}} \\
				& = E_m \iprod{\psii{m}}{\psik} + \sum_n b_n \left( \beta_{nm} + \sum_{\vetor R\ne \vetor 0} e^{i\vetor k\cdot\vetor R} \gamma_{nm}(\vetor R) \right)
	\end{align*}
	
	Assim, aplicando esses resultados no problema de autovalores obtemos 
	\begin{multline*}
	E_m\iprod{\psii{m}}{\psik} + \sum_n b_n\left( \beta_{nm} + \sum_{\vetor R\ne\vetor 0}e^{i\vetor k\cdot\vetor R}\,\gamma_{nm}(\vetor R)\right) = \\
	  = E(\vetor k)\,\iprod{\psii{m}}{\psik}.
	\end{multline*}
	
	Finalmente, utilizando $\iprod{\psii{n}}{\psii{m}}=\delta_{nm}$ e $\iprod{\psii{m}}{\psiiR{n}}=0$ obtemos a equação de autovalores procurada:	
	\begin{equation}\label{eq:eq-secular}
		\left[E(\vetor k) - E_m\right]\,b_m = \sum_n b_n \left( \beta_{nm} + \sum_{\vetor R\ne \vetor 0} e^{i\vetor k\cdot \vetor R} \gamma_{nm}(\vetor R) \right)
	\end{equation}

	Agora, para obtermos a matriz $4\times4$ explicitamente, tomamos $\beta_{nm}=0$ por simplicidade e separamos a soma nos termos $n=m$ e $n\ne m$:
	$$
	\left\{\sum_{\vetor R\ne \vetor 0} e^{i\vetor k\cdot \vetor R} \gamma_{mm}(\vetor R) - \left[ E(\vetor k)- E_m \right]\right\} b_m
	+ \sum_{n\ne m}^4 b_n \sum_{\vetor R\ne \vetor 0} e^{i\vetor k\cdot \vetor R} \gamma_{nm}(\vetor R) = 0
	$$

	Fazendo $\gamma_{nm} = \sum_{\vetor R\ne \vetor 0} e^{i\vetor k\cdot\vetor R}\gamma_{nm}(\vetor R)$ para tornar a equação mais compacta, encontramos
	$$
	\left\{\gamma_{mm} - \left[ E(\vetor k)- E_m \right]\right\} b_m + \sum_{n\ne m}^4 b_n \gamma_{nm} = 0
	$$

	Em notação matricial, 
	{\small
	$$
	\left(\begin{matrix}
		\gamma_{11} - \left[ E(\vetor k)- E_1 \right]	&	\gamma_{21}	& \gamma_{31} & \gamma_{41} \\
		\gamma_{12}	& \gamma_{22} - \left[ E(\vetor k)- E_2 \right] & \gamma_{32}	&	\gamma_{42} \\
		\gamma_{13}	&	\gamma_{23}	& \gamma_{33} - \left[ E(\vetor k)- E_3 \right]	&	\gamma_{43} \\
		\gamma_{14}	&	\gamma_{24}	&	\gamma_{34}	& \gamma_{44} - \left[ E(\vetor k)- E_4 \right]
	\end{matrix}\right)
	\left(\begin{matrix}
		b_1 \\ b_2 \\b_3 \\ b_4
	\end{matrix}\right) = 0
	$$}

	No caso de uma rede cúbica simples, cujos primeiros vizinhos (estamos ignorando as interações entre vizinhos mais distantes) estão localizados em
	$$
	\vetor R_{1,2} = \pm a(1,0,0)	\qquad \vetor R_{3,4} = \pm a(0,1,0) \qquad \vetor R_{5,6} = \pm a(0,0,1)
	$$

	Fazendo $\vetor k= (k_x,k_y,k_z)$ e utilizando a simetria $\gamma_{nm}(\vetor R) = \gamma_{nm}(-\vetor R)$,
	$$
	\gamma_{nm} = 2\left[\gamma_{nm}(\vetor R_1)\cos(k_xa) + \gamma_{nm}(\vetor R_3)\cos(k_ya) + \gamma_{nm}(\vetor R_5)\cos(k_za)\right]
	$$
	
	Observando que $E_1=E_{s0}$ e $E_2=E_3=E_4=E_{p0}$ e que $\gamma_{nm}$ é equivalente às integrais apresentadas no enunciado da questão, a equação matricial anterior fica completamente definida.

	\textbf{Exercício 10:} \textbf{(a)} Utilizando a equação~\eqref{eq:eq-secular} para $m=s$,
	$$
	E(\vetor k) - E_s = \beta_ss + e^{i\,a\,k}\gamma_{ss}(\vetor R_1) + e^{-i\,a\,k}\gamma_{ss}(\vetor R_2)
	$$
	
	Como $\vetor R_1 = -\vetor R_2$, podemos escrever $\gamma_{ss}(\vetor R_1) = \gamma_{ss}(\vetor R_2) = \gamma_{ss}$. Com isto, a expressão acima se torna
	$$
	E(\vetor k) = E_s + \beta_{ss} + 2\gamma_{ss}(\vetor R_1)\,\cos(ak)
	$$
	e a figura $5a$ ilustra o seu gráfico para $k\in[0,\pi/a]$.
	
	\textbf{(b)} A massa efetiva desta estrutura é
	$$
	\frac{1}{m^*} = \frac{1}{\hbar^2}\,\frac{\partial^2E}{\partial k^2} = \frac{2\,\gamma_{ss}\,a^2\,\cos(ak)}{\hbar^2}
		\quad\Rightarrow\quad m^* = \frac{\hbar^2}{2\,\gamma_{ss}\,a^2\,\cos(ak)}
	$$
	e a figura $5b$ ilustra o seu gráfico para $k\in[0,\pi/a]$.
