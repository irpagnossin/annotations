\begin{quotation}
{\color{blue}{\bf 6.}}
{\it	Usando a unidade astronômica como unidade de distância, anos como a unidade de tempo e a massa do Sol como a unidade de massa, a constante
	$k$ da {\it 3$^{\underline{a}}$ Lei de Kepler} vale $1,0$. Nestas unidades, qual é o valor da constante de gravitação $G$ de Newton?}
\end{quotation}

\vspace{5mm}

	A {\it 3$^{\underline{a}}$ Lei de Kepler} é
	$$
	P^2=ka^3,\;\;\;\;\; k={4\pi^2\over GM}
	$$

	onde $M$ é uma massa qualquer\footnote{Aqui não nos interessa ficar definindo qual massa é de qual planeta. Precisamos apenas saber que
$M$ representa uma massa.}. Se 
	$$
	[T]=\mathrm{anos},\;\;\;\; [M]=M_{\bigodot}\;\;\;\; e \;\;\;\;[D]=\mathrm{UA}
	$$  

	então 
	$$
	k=1={4\pi^2\over GM}={4\pi^2\over 1\cdot G}\Rightarrow G=4\pi^2\mathrm{UA^3\over M_{\bigodot}anos^2}
	$$
